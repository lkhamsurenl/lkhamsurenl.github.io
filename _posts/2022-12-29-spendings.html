# Add categories to fidelity with basic inference
def add_category(desc):
    if (
        "WHOLEFDS" in desc or 
        "TARGET" in desc or 
        "LONGS" in desc or 
        "DON QUIJOTE" in desc or 
        "ABC" in desc or
        "WAIKIKI OLD COUNTRY" in desc or
        "SAFEWAY" in desc or
        "DOWN TO EARTH ORGAN" in desc or
        "FOODLAND" in desc
    ): 
        return "Groceries"
    if "MORNING BREW" in desc or "MUSUBI CAFE" in desc:
        return "Food & Drink"
    if "CUTTER MAZDA" in desc:
        return "Automotive"
    if "TEXACO" in desc or "SHELL OIL" in desc or "AIM DIAMOND" in desc:
        return "Gas"
    if "WATANABE FLORAL" in desc:
        return "Shopping"
    
    return "UNK"

fidelity_df["category"] = fidelity_df["description"].apply(add_category)

# preprocess data with additional information such as type, month:
df["transaction_date"] = pd.to_datetime(df["transaction_date"])
df['month'] = df['transaction_date'].to_numpy().astype('datetime64[M]')
df['year'] = df['transaction_date'].dt.year
df = (
    df
    .query("amount < 0")
    .assign(amount = lambda r: np.abs(r["amount"]))
)
